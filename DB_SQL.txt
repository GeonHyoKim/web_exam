DROP DATABASE IF EXISTS web_exam;

CREATE DATABASE web_exam;

USE web_exam;

SELECT * FROM `member`
SELECT * FROM article
SELECT * FROM reply

DROP TABLE `member`
DROP TABLE article
DROP TABLE reply

CREATE TABLE `member`(
	id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT
	, email VARCHAR(100)
	, loginPw VARCHAR(20)
	, `name` VARCHAR(20)

)

CREATE TABLE article(
	id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT
	, title VARCHAR(20)
	, `body` VARCHAR(50)
	, memberId INT UNSIGNED

)

CREATE TABLE reply(
	id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT
	, regDate DATETIME NOT NULL
	, articleId INT UNSIGNED
	, reply VARCHAR(50)
	, memberId INT UNSIGNED

)

CREATE TABLE reply(
	id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT
	, regDate DATETIME NOT NULL
	, memberId INT UNSIGNED NOT NULL
	, relId INT UNSIGNED NOT NULL
	, `body` TEXT NOT NULL
);
